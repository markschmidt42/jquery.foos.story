How to use

Probably easiest to just view source of this page... But if you want, you can try to follow these awesome instructions :)

Add your js references and add a div

Name it whatever you want. 
<div class="game-story-container"></div>

<script src="https://ajax.googleapis.com/ajax/libs/jquery/1.12.4/jquery.min.js"></script>
<script src="js/jquery.foos.story.js"></script>

Add some js code (probably onready)

var $gs = $('.game-story-container');

$gs.gameStory(); // initializes the gamestory div
Advanced settings: 
// override defaults (great for testing)
$gs.gameStory({
	 gameTimeSeconds: (1 * 60) /// just the "default" game time, it will be smart
	,startingTime: 45 /// testing, just starts the game of at 45 seconds
	,autoExtendTimerBy: .50
	,debug: true
});

Add some js to your actions, there are 3 methods to use

$gs.gameStory('start');

$gs.gameStory('score', 'yellow defense', 'score label');

$gs.gameStory('stop');

This example:

$(function() { 

	var $gs = $('.game-story-container');

	// not doing default settings for this test
	//$gs.gameStory(); // uses default settings (5 minute default game, ...)

	$gs.gameStory({
		gameTimeSeconds: (1 * 60) // just the "default" game time, it will be smart and extend as needed
	});
	

	$('#start').on('click', function() { 
		$gs.gameStory('start');
	});
	
	$('#scoreYellowD').on('click', function() { 
		$gs.gameStory('score', 'yellow defense', 'MS'); // label could be 'D'
	});

	$('#scoreYellowO').on('click', function() { 
		$gs.gameStory('score', 'yellow offense', 'CH'); // could be 'O'
	});

	$('#scoreBlackD').on('click', function() { 
		$gs.gameStory('score', 'black defense', 'IT'); // could be 'D'
	});

	$('#scoreBlackO').on('click', function() { 
		$gs.gameStory('score', 'black offense', 'TC'); // could be 'O'
	});

	$('#stop').on('click', function() { 
		$gs.gameStory('stop');
	});

});
